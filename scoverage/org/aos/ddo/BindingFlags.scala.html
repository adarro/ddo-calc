23 /** Stores the binding status of an item.
24   *
25   * As this is basically a database / analysis tool, it is primarily for identifying if
26   * and when any binding may occur, which is useful when attempting to acquire said item.
27   */

29     override val entryName: String,

31     event: BindingEvent) extends EnumEntry with Abbreviation with DefaultValue[BindingFlags] {
32   val abbr: String = entryName
33   override lazy val default: Option[Unbound.type] = BindingFlags.default
34 }
35 /** Distinct value of binding options.
36   */
37 object BindingFlags extends Enum[BindingFlags] with DefaultValue[BindingFlags] with LazyLogging {
38   /** Attempts to mangle an Enumeration using free text.
39     *
40     * There is no extensive rule set, but DDO commonly using acronyms such as BTC,
41     * so we are translating Binds|Bound to Account to BTA etc.
42     *
43     * @note
44     * First catches 'Unbound'
45     * Next Attempts to extract abbreviation using space or case to separate words
46     *
47     * Finally defaults to None
48     *
49     * If you have the exact words without spaces, it may be more performant to use the 'withName' variant.
50     * NOTE: This enumeration is keyed by Acronym, so BTCoE will match, where 'Bound to Character On Equip' will fail.
51     */
52   def fromWords(words: Option[String]): Option[BindingFlags] = {
53     words match {
54       case Some(x) if x.equalsIgnoreCase(BindingFlags.Unbound.toFullWord()) =>
55         Some(BindingFlags.Unbound)
56       case Some(x) => x.wordsToAcronym match {
57         case Some(abbr) =>
58           BindingFlags.values.find { x => x.abbr.equalsIgnoreCase(abbr) } match {
59             case Some(opt) => Some(opt)
60             case _ =>
61               logger.warn(s"No Matching Value found for $words, defaulting")
62               None
63           }
64         case _ => None
65       }
66     }
67   }
68 
69   /** @see [[org.aos.ddo.BindingFlags.fromWords(#Option[String])]]
70     */
71   def fromWords(words: String): Option[BindingFlags] = fromWords(Option(words))
72 
73   /** Returns the default binding status (BindingFlags.Unbound)
74     */
75   override lazy val default: Option[Unbound.type] = Some(BindingFlags.Unbound)
76   val values = findValues
77   case object Unbound extends BindingFlags("Unbound", BindingStatus.Unbound, BindingEvent.None) {
78     def toFullWord(): String = "Unbound"
79   }
80   case object BoundToAccountOnAcquire extends BindingFlags("BTAoA", BindingStatus.BindsToAccount, BindingEvent.OnAcquire) {
81     def toFullWord(): String = "Bound To Account On Acquire"
82   }
83   case object BoundToCharacterOnAcquire extends BindingFlags("BTCoA", BindingStatus.BindsToAccount, BindingEvent.OnAcquire) {
84     def toFullWord(): String = "Bound To Character On Acquire"
85   }
86   case object BoundToAccount extends BindingFlags("BTA", BindingStatus.BindsToAccount, BindingEvent.None) {
87     def toFullWord(): String = "Bound To Account"
88   }

91   }
