63 trait Weapon extends Item  with MetaData with HandedWeapon {
64 
65   /** Proficiency Class
66     * Equipping a item without having the corresponding
67     * proficiency may incur penalties
68     *
69     * i.e. Repeating Heavy Crossbows require Martial Weapon Proficiency: Repeating Heavy Crossbows
70     * (certain requirements may be bypassed or temporarily granted via race / class enhancements or
71     * spells such as Tensor's Transformation
72     */
73   val proficiency: Option[ProficiencyClass]
74 
75   // Damage and Type: 1d6 Pierce
76   val damage: Option[DamageInfo]
77   // Critical threat range 20/x3
78   val critical: Option[CriticalProfile]
79 
80   /** Weapon Type
81     *
82     * Weapon Type represents a general class of weapons, such as Ranged Weapons for bows, thrown for
83     * small projectiles, etc.
84  *
85     * @see [[org.aos.ddo.model.item.weapon.WeaponType]]
86     */
87   val weaponType: Option[WeaponType]
88 
89   val weaponCategory: Option[WeaponCategory]
90 
91   /** Handedness
92     * Represents how a weapon can be used / equipped. i.e. Daggers (One Handed, Longbow  two handed)
93     *
94     * @note DDO Wiki seems a bit inconsistent.  [[http://ddowiki.com/page/Composite_longbow Composite Longbow]] lists
95     *       handedness as 'ranged' while [[http://ddowiki.com/page/Item:Bow_of_the_Silver_Flame Bow of the Silver Flame]] leaves this blank.
96     *       For our purposes, the handedness will likely be 'twohanded', and matching against the [[org.aos.ddo.model.item.weapon.WeaponType]] for Ranged should be
97     *       enough to allow the UI to display either or none and understand that using a bow prevents equipping something else
98     *       in the off hand.
99     *
100     * @todo Likely need to test this for logic / integration
101     */
102   val handedness: List[Handedness]
103 
104   /** Default Modifiers
105     * Dertermines the ability stat used for attack or damage bonuses.
106     * Typically this will be STR for melee / bows to Hit and / or damage.
107     * Weapon finesse and similar may change this to another stat such as DEX
108     * or INT for Insightful Weapons.
109     */
110   val damageModifier: List[Attributes]
111 
112   /** @see [[damageModifier]]
113     */
114   val attackModifier: List[Attributes]
115 
116   val upgradeable: UpgradeInfo
117 
118   //  Named items: Category:Composite Shortbows
119 
120   /** handwrap example
121     * Base Value: None
122     * Weight: None
123     * Proficiency Class: Simple weapons
124     * Damage and Type: 1d3 Bludgeon (Halflings: 1d2 Bludgeon)
125     * Critical threat range 20/x2
126     * Weapon Type: Bludgeoning weapons
127     * Handedness: One-handed weapon
128     * Default Modifiers
129     * Damage: STR
130     * Attack: STR
131     * Durability: Not applicable
132     * Made from: Flesh
133     * Hardness: N/A
134     * Named items: Category:Handwraps
135     */
136 
137   /** restricts the slot to main hand or off hand.
138     * Override this to restrict further for two handed or other special cases
139     */
140   def allowedWearLocationFlags: Int = WearLocation.MainHand.bitValue | WearLocation.OffHand.bitValue
141 }
142 
143 /** Basic trait for allowing an item to be wielded using one or more hand slots.
144   *
145   * TODO: +allowedWearLocations needs to block off-hand for two-handed melee / bows etc
146   * TODO: Bows / x-bows eqiup to main hand, need a plug or otherwise to 'occupy' the off-hand slot
147   *
148   */
149 trait HandedWeapon extends Wearable
150 
151 trait OneHandedWeapon extends HandedWeapon {
152   lazy val allowedWearLocations = {
153     WearLocation.fromMask(
154       WearLocation.MainHand.bitValue | WearLocation.OffHand.bitValue) match {
155       case Some(x) => x
156       case _       => Nil
157     }
158   }
159 }
160 
161 trait TwoHandedWeapon extends HandedWeapon {
162   lazy val allowedWearLocations = {
163     WearLocation.fromMask(WearLocation.MainHand.bitValue) match {
164       case Some(x) => x
165       case _       => List[WearLocation with Product with Serializable]()
166     }
167   }
168 }
